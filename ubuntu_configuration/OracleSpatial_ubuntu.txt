
Oracle Spatial in Mapserver activieren:
	http://download.osgeo.org/mapserver/
1- download and extract mapserver source and change directory:
	sudo wget http://download.osgeo.org/mapserver/mapserver-7.6.4.tar.gz
	sudo tar xvf mapserver-7.0.2.tar.gz
	sudo cd mapserver-7.0.2/

2- install cmake
	sudo apt-get install cmake
	
3- make build directory and change to it [~/mapserver-7.0.2/build]:
	mkdir build
	cd build	
	
4-install compile requirement
	sudo apt-get install libfreetype6-dev 
	sudo apt-get install libproj-dev
	sudo apt-get install libfribidi-dev
	sudo apt-get install libharfbuzz-dev
	sudo apt-get install libcairo-dev
	sudo apt-get install libgdal-dev
	sudo apt-get install libgpg-dev
	sudo apt-get install libaio1

5- 	change in FindOracle.cmake 
	
	  find_path(ORACLE_INCLUDE_DIR
		NAMES oci.h
		PATHS

		/opt/oracle/instantclient/sdk/include)
		#${ORACLE_HOME}/rdbms/public
		#${ORACLE_HOME}/include
		#${ORACLE_HOME}/sdk/include  # Oracle SDK
		#${ORACLE_HOME}/OCI/include) # Oracle XE on Windows

	  set(ORACLE_OCI_NAMES clntsh libclntsh oci)
	  set(ORACLE_NNZ_NAMES nnz10 libnnz10 nnz11 libnnz11 nnz12 libnnz12 nnz18 libnnz18 ociw32 nnz19 libnnz19)
	  set(ORACLE_OCCI_NAMES libocci occi oraocci10 oraocci11 oraocci12)

	  set(ORACLE_LIB_DIR 
		/opt/oracle/instantclient)
	   # ${ORACLE_HOME}
	   # ${ORACLE_HOME}/lib
	   # ${ORACLE_HOME}/sdk/lib       # Oracle SDK
	   # ${ORACLE_HOME}/sdk/lib/msvc
	   # ${ORACLE_HOME}/OCI/lib/msvc) # Oracle XE on Windows
	
	sudo cmake .. -DCMAKE_PREFIX_PATH=/opt/oracle/instantclient:/opt/oracle/instantclient/sdk/include: -DWITH_ORACLESPATIAL=ON -DWITH_ORACLE_PLUGIN=0 -DWITH_FRIBIDI=ON -DWITH_HARFBUZZ=ON -DWITH_RSVG=0
	sudo make
	sudo make install

6-  mapserv -v
	MapServer version 7.6.4 OUTPUT=PNG OUTPUT=JPEG SUPPORTS=PROJ SUPPORTS=AGG SUPPORTS=FREETYPE SUPPORTS=CAIRO SUPPORTS=ICONV 
	SUPPORTS=FRIBIDI SUPPORTS=WMS_SERVER SUPPORTS=WFS_SERVER SUPPORTS=WCS_SERVER SUPPORTS=FASTCGI SUPPORTS=GEOS 
	SUPPORTS=POINT_Z_M INPUT=JPEG INPUT=POSTGIS INPUT=ORACLESPATIAL INPUT=OGR INPUT=GDAL INPUT=SHAPEFILE

How to build swig on ubuntu?
	git clone https://github.com/swig/swig.git
	cd swig
	sudo apt-get install bison flex
	./autogen.sh
	./configure
	make
	sudo make install
